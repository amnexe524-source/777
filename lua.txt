-- 🖱 Draggable UI
---------------------------------------------------------
local dragging, dragInput, dragStart, startPos
MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true; dragStart = input.Position; startPos = MainFrame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then dragging = false end
        end)
    end
end)
MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)
UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

-- ⚙️ Timer Controls Logic
---------------------------------------------------------
local function applyNewTimer()
    local val = tonumber(MinutesInput.Text)
    if val and val >= 1 and val <= 999 then
        timerMinutes = val
        countdownTime = timerMinutes * 60
        local mins = math.floor(countdownTime / 60)
        local secs = countdownTime % 60
        TimerLabelUI.Text = string.format("%02d:%02d", mins, secs)
        TimerLabelUI.TextColor3 = Color3.fromRGB(0, 255, 150)
        print("⏱ Timer set to", timerMinutes, "minutes")
    else
        MinutesInput.Text = tostring(timerMinutes)
    end
end

MinusBtn.MouseButton1Click:Connect(function()
    local val = tonumber(MinutesInput.Text) or timerMinutes
    val = math.max(1, val - 1)
    MinutesInput.Text = tostring(val)
end)

PlusBtn.MouseButton1Click:Connect(function()
    local val = tonumber(MinutesInput.Text) or timerMinutes
    val = math.min(999, val + 1)
    MinutesInput.Text = tostring(val)
end)

ApplyBtn.MouseButton1Click:Connect(function()
    if not isResetting then
        applyNewTimer()
        -- Flash feedback
        ApplyBtn.BackgroundColor3 = Color3.fromRGB(0, 200, 100)
        task.delay(0.3, function()
            ApplyBtn.BackgroundColor3 = Color3.fromRGB(0, 140, 80)
        end)
    end
end)

MinutesInput.FocusLost:Connect(function(enterPressed)
    if enterPressed then applyNewTimer() end
end)

-- ⚙️ Core Functions

-- อัปเดต UI Status
local function updateStatus(text, color)
    StatusLabel.Text = text
    if color then StatusLabel.TextColor3 = color end
end

-- ถอด Tool ชั่วคราว
local function unequipTool(char)
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if hum then hum:UnequipTools() end
end

-- สวมใส่ Tool
local function autoEquip(char)
    if not char then return end
    local Humanoid = char:FindFirstChildOfClass("Humanoid")
    if not Humanoid or Humanoid.Health <= 0 then return end
    if not char:FindFirstChildOfClass("Tool") then
        local Backpack = LocalPlayer:FindFirstChild("Backpack")
        if Backpack then
            local targetTool = Backpack:FindFirstChild("Ice Queen") or Backpack:FindFirstChildOfClass("Tool")
            if targetTool then Humanoid:EquipTool(targetTool) end
        end
    end
end

-- เดินกลับจุดเริ่ม (ถอด Tool ขณะเดิน, ถือคืนเมื่อถึง)
local function autoReturn(char)
    if not char or not savedPosition then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hrp or not hum or hum.Health <= 0 then return end

    local distance = (hrp.Position - savedPosition).Magnitude
    if distance > 5 then
        if not isWalking then
            isWalking = true
            pcall(unequipTool, char)
            updateStatus("🚶 Walking to spawn...", Color3.fromRGB(255, 200, 50))
        end
        hum:MoveTo(savedPosition)
    else
        if isWalking then
            isWalking = false
            updateStatus("⚡ Running  •  Cycle: " .. cycleCount, Color3.fromRGB(140, 140, 160))
            -- ถึงจุดแล้ว เริ่ม equip ใหม่
            task.spawn(function()
                task.wait(0.3)
                pcall(autoEquip, LocalPlayer.Character)
            end)
        end
    end
end

-- รีเซ็ต 5 ครั้ง + Webhook (ไม่บอกชื่อผู้ใช้)
local function triggerResets()
    isResetting = true
    TimerLabelUI.TextColor3 = Color3.fromRGB(255, 80, 80)
    TimerLabelUI.Text = "RESET"
    StatusDot.BackgroundColor3 = Color3.fromRGB(255, 80, 80)
    updateStatus("💀 Resetting (5x)...", Color3.fromRGB(255, 80, 80))

    pcall(unequipTool, LocalPlayer.Character)

    for i = 1, 5 do
        local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
        local hum = char:WaitForChild("Humanoid", 5)
        if hum then hum.Health = 0 end
        LocalPlayer.CharacterAdded:Wait()
        task.wait(2.7)
    end

    cycleCount = cycleCount + 1

    -- Webhook ไม่มีชื่อผู้ใช้
    if requestFunc then
        local message = string.format(
            "✅ **[AUTO LUA]** Reset 5 ครั้ง สำเร็จ! (รอบที่ %d) • Timer: %d นาที",
            cycleCount, timerMinutes
        )
        pcall(function()
            requestFunc({
                Url = WebhookURL,
                Method = "POST",
                Headers = { ["Content-Type"] = "application/json" },
                Body = HttpService:JSONEncode({ ["content"] = message })
            })
        end)
    end

    -- รีเซ็ตค่า
    TimerLabelUI.TextColor3 = Color3.fromRGB(0, 255, 150)
    StatusDot.BackgroundColor3 = Color3.fromRGB(0, 255, 120)
    countdownTime = timerMinutes * 60
    isWalking = false
    isResetting = false
    updateStatus("⚡ Running  •  Cycle: " .. cycleCount, Color3.fromRGB(140, 140, 160))
end

---------------------------------------------------------
-- 🔄 Main Loops
---------------------------------------------------------

-- Loop: Equip & Walk (0.5s)
task.spawn(function()
    while task.wait(0.5) do
        if not isResetting then
            local char = LocalPlayer.Character
            pcall(autoReturn, char)
            -- Equip เฉพาะเมื่อไม่ได้กำลังเดิน
            if not isWalking then
                pcall(autoEquip, char)
            end
        end
    end
end)

-- Loop: Countdown (1s)
task.spawn(function()
    while task.wait(1) do
        if not isResetting then
            countdownTime = math.max(0, countdownTime - 1)
            local mins = math.floor(countdownTime / 60)
            local secs = countdownTime % 60
            TimerLabelUI.Text = string.format("%02d:%02d", mins, secs)

            -- เปลี่ยนสีเมื่อใกล้หมดเวลา
            if countdownTime <= 60 then
                TimerLabelUI.TextColor3 = Color3.fromRGB(255, 120, 50)
            elseif countdownTime <= 180 then
                TimerLabelUI.TextColor3 = Color3.fromRGB(255, 220, 0)
            else
                TimerLabelUI.TextColor3 = Color3.fromRGB(0, 255, 150)
            end

            if countdownTime <= 0 then
                task.spawn(triggerResets)
            end
        end
    end
end)
